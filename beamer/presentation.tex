%%% NEW COMMANDS

%% Easy comments

\ifdefined\showtodos
  % base function for comments, which will be hidden if showtodos is not defined
  \newcommand\easycomment[3][]{%
    #1\ignorespaces%
    \ifmmode%
      \text{#2\ignorespaces#3}%
    \else%
      {#2 \ignorespaces#3}%
    \fi%
  }
  % Text in \correct will be shown as normal text if showtodos is not defined
  \newcommand\correct[2][]{%
    #1\ignorespaces%
    \ifmmode%
      \text{\color[rgb]{0.9,0.4,0}\textsc{Correct}: \ignorespaces#2}%
    \else%
      {\color[rgb]{0.9,0.4,0}\textsc{Correct}: \ignorespaces#2}%
    \fi%
  }
\else
  \newcommand\easycomment[3][]{}
  \newcommand\correct[2][]{#2}
\fi
% Text in todo will be hidden if showtodos is not defined
\newcommand\todo[2][]{\easycomment[#1]{\small\color[rgb]{1,0,0}\textsc{Todo}: }{#2}}
\newcommand\question[2][]{\easycomment[#1]{\color{green}\textsc{Question}: }{#2}}
